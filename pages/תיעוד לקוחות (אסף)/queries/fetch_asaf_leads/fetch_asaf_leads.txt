SELECT * FROM asaf_sales_view
WHERE ('{{Input_Search.text}}' = '' OR שם_לקוח ILIKE '%' || '{{Input_Search.text}}' || '%' OR תעודת_זהות ILIKE '%' || '{{Input_Search.text}}' || '%')
AND ('{{Select_Status.selectedOptionValue}}' = '' OR סטטוס = '{{Select_Status.selectedOptionValue}}')
AND ('{{Select_Month.selectedOptionValue}}' = '' OR TO_CHAR(תאריך, 'MM/YYYY') = '{{Select_Month.selectedOptionValue}}')
ORDER BY תאריך DESC
LIMIT 50
