SELECT
    COUNT(*) AS total_leads,
    COUNT(CASE WHEN closing_amount > 0 THEN 1 END) AS closings,
    COALESCE(SUM(closing_amount), 0) AS total_closing,
    COALESCE(SUM(agent_appointment_amount), 0) AS total_appointment,
    COUNT(CASE WHEN harel_agent IS NOT NULL AND harel_agent != '' THEN 1 END) AS harel_leads,
    COUNT(CASE WHEN phoenix_agent IS NOT NULL AND phoenix_agent != '' THEN 1 END) AS phoenix_leads
FROM adi_leads_view
WHERE ({{!Select_Month.selectedOptionValue || Select_Month.selectedOptionValue === ''}}
     OR TO_CHAR(תאריך, 'MM/YYYY') = {{Select_Month.selectedOptionValue}})
