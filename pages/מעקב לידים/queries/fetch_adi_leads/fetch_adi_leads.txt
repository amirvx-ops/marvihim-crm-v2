SELECT * FROM adi_leads_view
WHERE CASE WHEN '{{Input_Search.text}}' = '' THEN TRUE
     ELSE שם_לקוח ILIKE '%' || '{{Input_Search.text}}' || '%'
        OR תעודת_זהות ILIKE '%' || '{{Input_Search.text}}' || '%'
     END
AND CASE WHEN '{{status.selectedOptionValue}}' = '' THEN TRUE
     ELSE סטטוס = '{{status.selectedOptionValue}}'
     END
AND CASE WHEN '{{insurance_company.selectedOptionValue}}' = '' THEN TRUE
     ELSE חברת_ביטוח = '{{insurance_company.selectedOptionValue}}'
     END
AND CASE WHEN '{{Select_Partner.selectedOptionValue}}' = '' THEN TRUE
     ELSE מקור_ליד = '{{Select_Partner.selectedOptionValue}}'
     END
ORDER BY תאריך DESC
LIMIT 50
